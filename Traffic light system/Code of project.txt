int ledR = 1;
int ledY = 2;
int ledG = 3;
void setup()
{
  pinMode(ledR, OUTPUT);
  pinMode(ledY, OUTPUT);
  pinMode(ledG, OUTPUT);
  
  
}

void loop()
{
  digitalWrite(ledR , HIGH);
  digitalWrite(ledY , LOW);
  digitalWrite(ledG , LOW);
  delay(5000);
  
  digitalWrite(ledR , LOW);
  digitalWrite(ledY , HIGH);
  digitalWrite(ledG , LOW);
  delay(2000);
  
  digitalWrite(ledR , LOW);
  digitalWrite(ledY , LOW);
  digitalWrite(ledG , HIGH);
  delay(5000);
}
